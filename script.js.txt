function bukaQuiz() {
    document.getElementById('konten').innerHTML = `
        <section style="animation: fadeIn 0.5s;">
            <h3>Kuis Interaktif</h3>
            <p>Manakah sel yang berfungsi untuk merombak atau menghancurkan sel-sel tulang yang sudah rusak?</p>
            <button onclick="cekJawaban(false)" class="btn-quiz">A. Osteoblas</button>
            <button onclick="cekJawaban(true)" class="btn-quiz">B. Osteoklas</button>
            <button onclick="cekJawaban(false)" class="btn-quiz">C. Osteosit</button>
            <div id="feedback" style="margin-top:20px; font-weight:bold;"></div>
        </section>
    `;
}

function cekJawaban(benar) {
    const feedback = document.getElementById('feedback');
    if(benar) {
        feedback.innerHTML = "<span style='color:green;'>✅ Benar! Osteoklas berfungsi merombak tulang.</span>";
    } else {
        feedback.innerHTML = "<span style='color:red;'>❌ Salah. Ayo baca materi lagi!</span>";
    }
}
